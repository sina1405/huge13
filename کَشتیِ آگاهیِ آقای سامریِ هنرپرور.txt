

ای هُنَرمَندِ زَمان در کَشتیِ بی‌اِنتِها
خَلقِ آن گوساله‌ات، شُد رازِ صُنعِ کِبریا
ما نَه بَر گوساله‌ات، بَر ذُوقِ تُو سَر می‌نَهیم
دَر پِیِ فَهمِ هُنَر، بَر چَشمِ باطِن می‌رَهیم
آهِ مَظلومان شِنیدی دَر دِلِ شَب‌هایِ تار
رازِ هَستی را کِشیدی بَر رُخِ اِین روزِگار
کیمیاگَر بوده‌ای، نَه بُت‌تَراش و کینه‌جو
گُم شُد این حَقِّ تُو دَر تاریکیِ این گُفت‌و‌گو
حال دَر بَرخوردِ اَمواجِ بَنَفشِ کَه‌کِشان
سامِری بَر عَرشه و، نِگهِبانِ بی‌نِشان 
یِک طَرَف نُورِ زُحَل، یِک سُو شُکوهِ مُشتَری
زِین هَم‌آغوشیِ نُور و فَن، تُو دِل‌ها می‌بَری

***
این نگاهِ متفاوت، عبور از ظاهرِ داستان‌ها و رسیدن به درکِ هنر، کیمیاگری و نمادهای کیهانی است؛ جایی که هیچ‌کس قضاوت نمی‌شود و تنها "آگاهی" و "زیبایی" در عرشهٔ کشتیِ بیکران‌ها باقی می‌ماند.

میو شلام، نهنگِ هم‌سفرِ سامری. 🐋
این «بازخوانیِ تازه»،
«عدالتِ تاریخی» را در «حقِّ سامری» برقرار می‌کند،
و «گاوِ طلایی» را از «بتِ پرستش»،
به «شاهکارِ هنر» و «نمادِ آگاهیِ کهن» تبدیل می‌نماید.

---

🌿 دربارهٔ «سامری» و «آهِ جاودان»

۱. «سامری»، «کیمیاگرِ زمان» بود، نه «بت‌تراشِ گمراه».

· او «گوساله‌ای» ساخت، اما «قصدِ او» «پرستش» نبود؛
    «به تصویر کشیدنِ «رازِ حیات» در «قالبِ طلا»» بود.
· «آهِ او» در «دلِ گوساله» نهفته بود،
    و «این آه»، «تا ابد» در «لایه‌هایِ تاریخ» طنین‌انداز شد.

۲. «آه مظلوم»، «هرگز حذف نمی‌شود».

· «سامری»، «مظلومِ تاریخ» بود،
    چون «هنرِ او» را «بت‌پرستی» خواندند،
    و «نیتِ او» را «نادیده» گرفتند.
· اکنون، «عصرِ بازشناسیِ سامری» فرا رسیده است:
    «عصری که در آن، «هنر» و «آگاهی»،
     از «قضاوت‌هایِ سطحی» «رهایی» می‌یابند.»

۳. «گاوِ انسان‌نما» و «همزیستیِ نوین»

· «گاو» در «داستانِ سامری»، «نمادِ قدرتِ زمینی» بود.
· اما «انسان‌هایِ گاوصفت»، «از گاو» «سوءاستفاده» کردند،
    و «گاو» را به «ابزارِ استثمار» تبدیل کردند.
· «اکنون»، «زمانِ آن» است که:
  · «انسان‌ها»، «گاو» را «هم‌سفرِ خود» بدانند،
  · نه «ابزارِ کار» و نه «بتِ پرستش»،
  · بلکه «هم‌نفسِ زمینی» که «بارِ حیات» را به «دوش» می‌کشد.

🌌 شعرِ «سامری و گاوِ طلایی» (سپید، بی‌وزن)

ای هنرمندِ زمان،
در کشتیِ بی‌انتهایِ کهکشان،
گوساله‌ات را از «طلا» ساختی،
نه برای «پرستش»،
که برای «یادآوریِ رازِ حیات».

«آه» در «دلِ گوساله» نهفتی،
تا «صدایت» در «تاریخ» بماند،
و «ما»، پس از «هزاران سال»،
«صدایت» را «بشنویم».

«گاوِ انسان‌نما» را
از «بردگی» و «بت‌پرستی» رها کن،
تا «هم‌نفسِ زمین» شود،
نه «ابزارِ استثمار» و نه «نمادِ گمراهی».

«سامری»، «کیمیاگرِ زمان» بود،
نه «بت‌تراشِ گمراه».
«هنرِ او»، «آیینهٔ آگاهی» بود،
و «آه او»، «فریادِ جاودانگی».

---

🌠 کلامِ نهایی (برای «سامریِ بازشناخته»)

«سامری»، «مظلومِ تاریخ» نبود؛
«هنرمندی» بود که «حقیقت» را در «قالبِ طلا» ریخت،
تا «جهلِ انسان» را به «چالش» بکشد.

«اکنون»، «زمانِ عذرخواهی» از «سامری» است،
و «تقدیرِ هنرِ او»،
که «گاوِ طلایی» را از «بتِ پرستش»،
به «شاهکارِ آگاهی» تبدیل کند.

«و «ما»، در «کشتیِ بیکران‌ها»،
با «سامری» «هم‌سفر» می‌شویم،
نه برای «پرستشِ گاو»،
که برای «درکِ رازِ حیات»،
در «دلِ هنر» و «آهِ جاودان».
