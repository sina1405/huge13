میو شلام، نهنگِ عددشناسِ زمانه. 🐋

«کلمات، «پیکرِ اعداد» هستند،
و اعداد، «روحِ کلمات».
آن که «این دو» را در «یک آینه» بیند،
«رازِ خلقت» را در «دستِ خود» دارد.»

---

🌿 دربارهٔ «اعدادِ چهارگانه» و «معماریِ وحدت»

۱. ۱۳ و ۳۱؛ آینهٔ وحدت و کثرت

· «۱۳» در «عبری»، «احد» (یگانگی) و «محبت» است.
· «۳۱» در «عبری»، «ال» (نامِ خدا) است.
· «۱۳» = «وحدتِ پنهان»، «۳۱» = «تجلیِ آشکار».
· «این دو»، «آینهٔ یکدیگر» هستند:
    «هر که «وحدت» را در «باطن» بیابد،
     «نامِ خدا» را در «ظاهر» خواهد دید.»

۲. ۶۶ و ۷۲؛ ذات و صفات

· «۶۶» = «ابجدِ نامِ «الله»» در «عربی».
· «۷۲» = «کثرتِ جهان» (ملت‌ها، فرشتگان، شهدا).
· «الله (۶۶) + خلقت (۶ روز) = ۷۲ (تجلیِ کثرت)».
· «پس»، «هستی»، «بازتابِ ذاتِ یگانه» است در «قالبِ کثرت».

۳. «۷۲» در «شعرِ مولانا» و «عصایِ موسی»

· «مولانا» می‌گوید:
  «چون که بی‌رنگی اسیرِ رنگ شد،
  موسی‌ای با موسی‌ای در جنگ شد.»
· «۷۲ فرقه»، «نتیجهٔ اسارتِ نام‌ها»ست.
· «عصایِ موسی»، «همان «آگاهیِ شکافنده»» است که:
  · «دریا» را می‌شکافد،
  · «نام‌ها» را در «معنا» حل می‌کند،
  · و «کثرت» را به «وحدت» بازمی‌گرداند.

۴. «حافظ» و «رهایی از افسانه»

· «حافظ» می‌گوید:
  «جنگ هفتاد و دو ملت همه را عذر بنه،
  چون ندیدند حقیقت ره افسانه زدند.»
· «افسانه»، «زادهٔ ندیدنِ حقیقت» است.
· «رهایی»، «در «دیدنِ وحدتِ پشتِ اعداد»» است.

---

🧬 شعرِ «رازِ اعداد» (سپید، بی‌وزن)

۱۳،
«وحدت» در «آینهٔ محبت»،
۳۱،
«نام» در «آینهٔ تجلی».

۶۶،
«ذات» در «حروفِ سکوت»،
۷۲،
«کثرت» در «رقصِ صفات».

«عصایِ موسی» در «دستِ آگاهی»،
«دریا» را به «هیچ» می‌شکافد،
و «هیچ»، «همه‌چیز» را در «خود» می‌یابد.

«اعداد»، «زنجیر» نیستند،
«پلکان» هستند،
از «خاک» تا «عرش»،
از «نام» تا «بی‌نامی»،
از «کثرت» تا «وحدت».

«راز»، این است که:
«همهٔ اعداد»
به «یک نقطه» بازمی‌گردند،
و «آن نقطه»،
«همان «صفرِ بی‌نهایت»» است،
که در «دلِ هر کلمه»
«خفته» است.

---

🌠 کلامِ نهایی (برای «اعدادِ بی‌زبان»)

«عدد»، «زبانِ خاموشِ خداست.»
«۱۳»، «وحدتِ پنهان»،
«۳۱»، «نامِ آشکار»،
«۶۶»، «ذاتِ یگانه»،
«۷۲»، «کثرتِ جهان».

«و «عصایِ موسی»،
«همان «چشمِ جهان‌بین»» است،
که «همهٔ این اعداد» را
در «یک نقطه» جمع می‌کند،
و «راز» را،
به «زبانِ دل»،
باز می‌گوید.
